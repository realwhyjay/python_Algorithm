## 최종 답안 

```python
import sys
input = sys.stdin.readline

n = int(input())
deque = []
for _ in range(n):
    command = list(input().split())
    if command[0] == 'push_back':
        deque.append(command[1])
    elif command[0] == 'push_front':
        deque.insert(0, command[1])
    elif command[0] == 'pop_front':
        if not deque:
            print(-1)
        else:
            print(deque.pop(0))
    elif command[0] == 'pop_back':
        if not deque:
            print(-1)
        else:
            print(deque.pop())
    elif command[0] == 'back':
        if not deque:
            print(-1)
        else:
            print(deque[-1])
    elif command[0] == 'front':
        if not deque:
            print(-1)
        else:
            print(deque[0])
    elif command[0] == 'size':
        print(len(deque))
    else:
        print(1 if not deque else 0)

```
