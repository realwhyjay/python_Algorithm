```python
def GCD(a, b):
  # 큰 수에서 작은 수를 나누어주어야하기 때문에, 입력 받은 수 중 뒤의 수가 더 큰 경우에는 위치를 바꿔준다
    if a < b:
        a, b = b, a
    if b == 0:
        return a
    else:
        return GCD(b, a % b)


t = int(input())

while t != 0:
    n = list(map(int, input().split()))
    sum = 0

    if n[0] != len(n)-1:
        break

    for i in range(1, len(n)-1):
        for j in range(i+1, len(n)):
            sum += GCD(n[i], n[j])
    print(sum)
    t -= 1

```
